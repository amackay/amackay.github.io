\( \sqrt{2} \notin \mathbb{Q} \)

\( ( (\sqrt{2} \in \mathbb{Q}) \Rightarrow \text{contradiction} ) \Rightarrow (\sqrt{2} \notin \mathbb{Q}) \)

\( (\sqrt{2} \in \mathbb{Q}) \Rightarrow \text{contradiction} \)

\( \text{assume } \sqrt{2} \in \mathbb{Q} \)

\( \text{let } n = \text{lowest_numerator}(\sqrt{2})\)

\( \text{let } d = \text{lowest_denominator}(\sqrt{2})\)

\( \text{then we have...} \)

\( n \text{ and } d \text{ have no common factors} \)

\(\frac{n}{d} = \sqrt{2}\)

\(2d^2 = n^2\)

\(\sqrt{2} = \frac{n}{d}\)

\(2 = (\frac{n}{d})^2\)

\(2 = \frac{n^2}{d^2}\)

\(2d^2 = n^2\)

\(n \text{ is even} \)
\( (n^2 \text{ is even}) \Rightarrow (n \text{ is even}) \)

\( \forall x. (x^2 \text{ is even}) \Rightarrow (x \text{ is even}) \)

\( \text{with } x = n \)

\(n^2 \text{ is even} \)

\(\therefore n \text{ is even} \)

\(d \text{ is even} \)

\(\text{contradiction}\)

\( n \text{ and } d \text{ have no common factors} \)

\( 2 \text{ is a factor of } n \)

\( n \text{ is even} \)

\( \therefore 2 \text{ is a factor of } n \)

\( 2 \text{ is a factor of } d \)

\( d \text{ is even} \)

\( \therefore 2 \text{ is a factor of } d \)

\(\therefore \text{contradiction}\)

\( \therefore \sqrt{2} \notin \mathbb{Q} \)
